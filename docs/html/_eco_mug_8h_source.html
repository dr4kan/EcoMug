34 class EMRandom {
35 private:
36  EMRandom() {
37  std::random_device rd;
38  std::mt19937 gen(rd());
39  std::uniform_int_distribution<uint64_t> dis(0, std::numeric_limits<uint64_t>::max());
40  s[0] = dis(gen);
41  s[1] = dis(gen);
42  };
43 
44  double GenerateRandomDouble() {
45  uint64_t x = next();
46  return to_double(x);
47  };
48 
49  int64_t rotl(const uint64_t x, int k) {
50  return (x << k) | (x >> (64 - k));
51  };
52 
53  uint64_t next() {
54  const uint64_t s0 = s[0];
55  uint64_t s1 = s[1];
56  const uint64_t result = s0 + s1;
57  s1 ^= s0;
58  s[0] = rotl(s0, 55) ^ s1 ^ (s1 << 14);
59  s[1] = rotl(s1, 36);
60  return result;
61  };
62 
63  double to_double(uint64_t x) {
64  const union { uint64_t i; double d; } u = { .i = UINT64_C(0x3FF) << 52 | x >> 12 };
65  return u.d - 1.0;
66  };
67 
68  uint64_t s[2];
69 };

296  void ComputeMaxSkyCylinder() {
297  if (mMinimumMomentum < 0.448) return;
298  double n = 2.856-0.655*log(mMinimumMomentum);
299  mMaxFuncSkyCylinder = 1600*pow(mMinimumMomentum+2.68, -3.175)*pow(mMinimumMomentum, 0.279)*pow(cos(0.6968), n+1)*sin(0.6968);
300  };

320  void Generate() {
321  bool accepted = false;
322  double r1t = 0.;
323  double r1m = 0.;
324  double r1p = 0.;
325  double r2 = 0.;
326  double n = 0.;
327  double ftheta = 0.;
328  double fphi = 0.;
329 
330  // Sky or cylinder generation
331  if (mGenMethod == Sky || mGenMethod == Cylinder) {
332  // Generation of the momentum and theta angle
333  while (!accepted) {
334  r1t = mRandom.GenerateRandomDouble();
335  r1m = mRandom.GenerateRandomDouble();
336  r2 = mRandom.GenerateRandomDouble();
337  mGenerationTheta = (mMaximumTheta - mMinimumTheta)*r1t + mMinimumTheta;
338  mGenerationMomentum = (mMaximumMomentum - mMinimumMomentum)*r1m + mMinimumMomentum;
339  n = 2.856-0.655*log(mGenerationMomentum);
340  if (n < 0.1) n = 0.1;
341 
342  if (mGenMethod == Sky) {
343  ftheta = 1600*pow(mGenerationMomentum+2.68, -3.175)*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n+1)*sin(mGenerationTheta);
344  if (mMaxFuncSkyCylinder*r2 < ftheta) accepted = true;
345  }
346 
347  if(mGenMethod == Cylinder) {
348  ftheta = 1600*pow(mGenerationMomentum+2.68, -3.175)*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n)*pow(sin(mGenerationTheta), 2);
349  if (mMaxFuncSkyCylinder*r2 < ftheta) accepted = true;
350  }
351  }
352  mGenerationTheta = M_PI - mGenerationTheta;
353 
354  // Generation of the position and phi angle
355  if (mGenMethod == Sky) {
356  GeneratePositionSky();
357  r1p = mRandom.GenerateRandomDouble();
358  mGenerationPhi = (mMaximumPhi - mMinimumPhi)*r1p + mMinimumPhi;
359  }
360  if (mGenMethod == Cylinder) {
361  double phi0 = GeneratePositionCylinder();
362  accepted = false;
363  while (!accepted) {
364  r1p = mRandom.GenerateRandomDouble();
365  r2 = mRandom.GenerateRandomDouble();
366  mGenerationPhi = (mMaximumPhi - mMinimumPhi)*r1p + mMinimumPhi;
367  fphi = fabs(cos(mGenerationPhi));
368  if (r2 < fphi) {
369  accepted = true;
370  }
371  }
372  mGenerationPhi = mGenerationPhi + phi0;
373  if (mGenerationPhi >= 2.*M_PI) mGenerationPhi -= 2.*M_PI;
374 
375  // Check if the muon is inward
376  if (sin(mGenerationTheta)*cos(mGenerationPhi)*mGenerationPosition[0] + sin(mGenerationTheta)*sin(mGenerationPhi)*mGenerationPosition[1] > 0) Generate();
377  }
378  }
379 
380  // Half-sphere generation
381  if (mGenMethod == HSphere) {
382  // Generation point on the half-sphere
383  double phi0 = 2.*M_PI*mRandom.GenerateRandomDouble();
384  double theta0 = 0.;
385  double r0t = 0.;
386  while (!accepted) {
387  r0t = mRandom.GenerateRandomDouble();
388  r1t = mRandom.GenerateRandomDouble();
389  r1m = mRandom.GenerateRandomDouble();
390  r1p = mRandom.GenerateRandomDouble();
391  r2 = mRandom.GenerateRandomDouble();
392  theta0 = acos(r0t);
393  mGenerationTheta = (mMaximumTheta - mMinimumTheta)*r1t + mMinimumTheta;
394  mGenerationPhi = (mMaximumPhi - mMinimumPhi)*r1p + mMinimumPhi;
395  mGenerationMomentum = (mMaximumMomentum - mMinimumMomentum)*r1m + mMinimumMomentum;
396  n = 2.856-0.655*log(mGenerationMomentum);
397  if (n < 0.1) n = 0.1;
398 
399  ftheta = 1600*pow(mGenerationMomentum+2.68, -3.175)*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n)*fabs(sin(mGenerationTheta)*sin(theta0)*cos(mGenerationPhi)+cos(mGenerationTheta)*cos(theta0))*sin(mGenerationTheta);
400  if (11*r2 < ftheta) accepted = true;
401  }
402 
403  mGenerationPosition[0] = mHSphereRadius*sin(theta0)*cos(phi0) + mHSphereCenterPosition[0];
404  mGenerationPosition[1] = mHSphereRadius*sin(theta0)*sin(phi0) + mHSphereCenterPosition[1];
405  mGenerationPosition[2] = mHSphereRadius*cos(theta0) + mHSphereCenterPosition[2];
406 
407  mGenerationTheta = M_PI - mGenerationTheta;
408  mGenerationPhi = mGenerationPhi + phi0;
409  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
410  mGenerationPhi += M_PI;
411  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
412  }
413  };
