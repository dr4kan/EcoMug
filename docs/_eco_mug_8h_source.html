33 class EMRandom {
34 public:
35  EMRandom() {
36  std::random_device rd;
37  std::mt19937 gen(rd());
38  std::uniform_int_distribution<uint64_t> dis(0, std::numeric_limits<uint64_t>::max());
39  s[0] = dis(gen);
40  s[1] = dis(gen);
41  };
42 
43  void SetSeed(uint64_t seed) {
44  s[0] = seed;
45  s[1] = seed;
46  };
47 
48  double GenerateRandomDouble() {
49  uint64_t x = next();
50  return to_double(x);
51  };
52 
53  double GenerateRandomDouble(double x1, double x2) {
54  return (x2-x1)*GenerateRandomDouble()+x1;
55  };
56 
57  int64_t rotl(const uint64_t x, int k) {
58  return (x << k) | (x >> (64 - k));
59  };
60 
61  uint64_t next() {
62  const uint64_t s0 = s[0];
63  uint64_t s1 = s[1];
64  const uint64_t result = s0 + s1;
65  s1 ^= s0;
66  s[0] = rotl(s0, 55) ^ s1 ^ (s1 << 14);
67  s[1] = rotl(s1, 36);
68  return result;
69  };
70 
71  double to_double(uint64_t x) {
72  const union { uint64_t i; double d; } u = { .i = UINT64_C(0x3FF) << 52 | x >> 12 };
73  return u.d - 1.0;
74  };
75 
76  uint64_t s[2];
77 };

316  return 1. - 8.534790171171021/pow(x + 2.68, 87./40.);

341 public:
345  void Generate() {
346  bool accepted = false;
347  double r2 = 0.;
348  double n = 0.;
349  double ftheta = 0.;
350  double fphi = 0.;
351 
352  // Sky or cylinder generation
353  if (mGenMethod == Sky || mGenMethod == Cylinder) {
354  // Generation of the momentum and theta angle
355  while (!accepted) {
356  r2 = mRandom.GenerateRandomDouble();
357  mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
358  mGenerationMomentum = GenerateMomentumF1();
359  n = 2.856-0.655*log(mGenerationMomentum);
360  if (n < 0.1) n = 0.1;
361 
362  if (mGenMethod == Sky) {
363  ftheta = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n+1)*sin(mGenerationTheta);
364  if (2797*r2 < ftheta) accepted = true;
365  }
366 
367  if(mGenMethod == Cylinder) {
368  ftheta = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n)*pow(sin(mGenerationTheta), 2);
369  if (4730*r2 < ftheta) accepted = true;
370  }
371  }
372  mGenerationTheta = M_PI - mGenerationTheta;
373 
374  // Generation of the position and phi angle
375  if (mGenMethod == Sky) {
376  GeneratePositionSky();
377  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
378  }
379  if (mGenMethod == Cylinder) {
380  double phi0 = GeneratePositionCylinder();
381  accepted = false;
382  while (!accepted) {
383  r2 = mRandom.GenerateRandomDouble();
384  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
385  fphi = fabs(cos(mGenerationPhi));
386  if (r2 < fphi) accepted = true;
387  }
388  mGenerationPhi = mGenerationPhi + phi0;
389  if (mGenerationPhi >= 2.*M_PI) mGenerationPhi -= 2.*M_PI;
390 
391  // Check if the muon is inward
392  if (sin(mGenerationTheta)*cos(mGenerationPhi)*mGenerationPosition[0] + sin(mGenerationTheta)*sin(mGenerationPhi)*mGenerationPosition[1] > 0) Generate();
393  }
394  }
395 
396  // Half-sphere generation
397  if (mGenMethod == HSphere) {
398  // Generation point on the half-sphere
399  double phi0 = 2.*M_PI*mRandom.GenerateRandomDouble();
400  double theta0 = 0.;
401  double r0t = 0.;
402  while (!accepted) {
403  r0t = mRandom.GenerateRandomDouble();
404  r2 = mRandom.GenerateRandomDouble();
405  theta0 = acos(r0t);
406  mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
407  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
408  mGenerationMomentum = GenerateMomentumF1();
409  n = 2.856-0.655*log(mGenerationMomentum);
410  if (n < 0.1) n = 0.1;
411 
412  ftheta = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), n)*(sin(mGenerationTheta)*sin(theta0)*cos(mGenerationPhi)+cos(mGenerationTheta)*cos(theta0))*sin(mGenerationTheta);
413  if (4891*r2 < ftheta) accepted = true;
414  }
415 
416  mGenerationPosition[0] = mHSphereRadius*sin(theta0)*cos(phi0) + mHSphereCenterPosition[0];
417  mGenerationPosition[1] = mHSphereRadius*sin(theta0)*sin(phi0) + mHSphereCenterPosition[1];
418  mGenerationPosition[2] = mHSphereRadius*cos(theta0) + mHSphereCenterPosition[2];
419 
420  mGenerationTheta = M_PI - mGenerationTheta;
421  mGenerationPhi = mGenerationPhi + phi0;
422  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
423 
424  mGenerationPhi += M_PI;
425  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
426  }
427 
428  // Generate the charge
429  mCharge = (mDiscDistC(mEngineC) == 0) ? 1 : -1;
430  };
